## 7. 整数反转

```python
class Solution:
    def reverse(self, x: int) -> int:
        res = 0
        if x < 0:
            num = -x
            while num > 0:
                lastOne = num % 10
                num = int(num / 10)
                res = res * 10 + lastOne
                if res > 2147483648:
                    return 0
            return -res
        else:
            num = x
            while num > 0:
                lastOne = num % 10
                num = int(num / 10)
                res = res * 10 + lastOne
                if res >= 2147483648:
                    return 0
            return res
```
