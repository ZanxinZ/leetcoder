class Solution {
    public int romanToInt(String s) {
        HashMap<Character,Integer> map = new HashMap<Character,Integer>();
        map.put('I',1);
        map.put('V',5);
        map.put('X',10);
        map.put('L',50);
        map.put('C',100);
        map.put('D',500);
        map.put('M',1000);
        char next;
        int sum=0;
        for(int i = 0;i < s.length();i++){
            char c = s.charAt(i);
            
            if(i == s.length()-1)next = ' ';
            else next = s.charAt(i+1);

            if(c == 'I' && next == 'V'){
                    sum += 4;
                    i++;
            }
            else if(c == 'I' && next == 'X'){
                sum += 9;
                i++;
            }
            else if(c == 'X' && next == 'L'){
                sum += 40;
                i++;
            }
            else if(c == 'X' && next =='C'){
                sum += 90;
                i++;
            }
            else if(c == 'C' && next == 'D'){
                sum += 400;
                i++;
            }
            else if(c == 'C' && next == 'M'){
                sum += 900;
                i++;
            }
            else{
                sum += map.get(new Character(c)).intValue();
            }
        }
        return sum;
    }
}
